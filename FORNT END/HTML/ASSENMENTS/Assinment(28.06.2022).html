if(result === 'prime'){
opt.classList.add('box-prime');
}
else if(result === 'even'){
opt.classList.add('box-even');
}
else{
opt.classList.add('box-odd');
}

select.appendChild(opt);
}

function isNumberCheck(input){
let flag = false;

for(let i=2;i<input;i++){
if(input%i===0){
flag = true;
break;
}
}

if(!flag && input >1){
return 'prime';
}
else if(input%2===0){
return 'even';
}
else{
return 'odd';
}
}
